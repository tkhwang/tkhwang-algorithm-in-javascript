```typescript
    const [ ROWS, COLS ] = [ grid.length, grid[0].length ];

    const directions = [ [1,0], [-1, 0], [0,1], [0, -1] ];
    const isValid = (r,c) => !(r < 0 || r >= ROWS || c < 0 || c >= COLS);

    const dfs = (r, c) => {
        for (const [ dR, dC ] of directions) {
            const newR = r + dR;
            const newC = c + dC;

            if (!isValid(newR, newC)) continue;
            // check validity

            dfs(newR, newC);
        }
    }
```
