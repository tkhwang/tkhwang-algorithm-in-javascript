```typescript
    const dfs = (node) => {
        if (!node) return;

        // pre-order
        console.log(`PRE: ${node.val}`);

        dfs(node.left);

        // in-order
        console.log(`IN: ${node.val}`);

        dfs(node.right);

        // post-order
        console.log(`OUT: ${node.val}`);
    }
```

### PRE-order :  Root -> Left Subtree -> Right Subtree

```
PRE: 4
PRE: 2
PRE: 1
PRE: 3
PRE: 6
```


### IN-order : Left Subtree -> Root -> Right Subtree

```
IN: 1
IN: 2
IN: 3
IN: 4
IN: 6
```



### POST-order : Left Subtree -> Right Subtree -> Root

```
OUT: 1
OUT: 3
OUT: 2
OUT: 6
OUT: 4
```
